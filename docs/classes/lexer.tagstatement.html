Class TagStatement

The tag statement parses multiline content inside an edge tag starting block.

const statement = new TagStatement(1)
statement.feed('@if(')
statement.feed('username')
statement.feed(')')

console.log(statement.props)
{
  name: 'if',
  jsArg: ' username ',
  raw: 'if(\nusername\n)'
}
